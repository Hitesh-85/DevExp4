nano install_nginx.yml


---
- name: Install and start NGINX
  hosts: local
  become: yes

  tasks:
    - name: Install nginx
      apt:
        name: nginx
        state: present
        update_cache: yes

    - name: Start nginx service
      service:
        name: nginx
        state: started
        enabled: yes


ansible-playbook -i hosts install_nginx.yml --syntax-check

sudo ansible-playbook -i hosts install_nginx.yml
